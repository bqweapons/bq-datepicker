デフォルト設定の日付ピッカー

表示フォーマット：NNNNyoM月D日（例：令和3年5月11日）
入力フォーマット：NNyMMDD （例：R30511）
htmlソース：
<div class="input-group col-md-3">
  <input type="text" class="form-control" id="sample1">
  <div class="input-group-append">
    <span class="input-group-text datepicker-button"><i class="far fa-calendar-alt"></i></span>
  </div>
</div>
JSソースコード：
document.querySelector("#sample1").datepicker();

JSでフォーマット設定

表示フォーマット：YYYY年MM月DD日（例：2021年5月11日）
入力フォーマット：YYYY/MM/DD （例：2021/05/11）
htmlソース：
<div class="input-group col-md-3">
  <input type="text" class="form-control" id="sample3">
  <div class="input-group-append">
    <span class="input-group-text datepicker-button"><i class="far fa-calendar-alt"></i></span>
  </div>
</div>
JSソースコード：
document.querySelector("#sample3").datepicker({
edit_format: "YYYY/MM/DD", // 編集時のフォーマット
month_format: "YYYY年MM月", // 年月フォーマット
display_format: "YYYY年MM月DD日", // 表示フォーマット
});

選択データよりフォーマット切替

htmlソース：
<input type="radio" value="0" id="format_NNyMMDD" name="radio_fromat" checked onclick="changeFormat(this.value)">
<label for="format_NNyMMDD" >
  和暦フォーマット
  <small>
    表示フォーマット：NNNNyoM月D日（例：令和3年5月11日）、
    入力フォーマット：NNyMMDD （例：R30511）
    </small>
</label>

<input type="radio" value="1" id="format_YYYYMMDD" name="radio_fromat" onclick="changeFormat(this.value)">
<label for="format_YYYYMMDD" >
  西暦フォーマット
  <small>
    表示フォーマット：YYYY年MM月DD日（例：令和3年5月11日）、
    入力フォーマット：YYYYMMDD （例：R30511）
  </small>
</label>

<div class="input-group col-md-3">
  <input type="text" class="form-control" id="sample4">
  <div class="input-group-append">
    <span class="input-group-text datepicker-button"><i class="far fa-calendar-alt"></i></span>
  </div>
</div>
JSソースコード：
<!-- 初期化処理 -->
document.querySelector("#sample4").datepicker();

<!-- 日付フォーマット変更処理 -->
function changeFormat(flg){
  let foramt = {};
  if (flg == "1") {
    foramt = {
      edit_format: "YYYYMMDD",              // 編集時のフォーマット
      month_format: "YYYY年MM月",           // 年月フォーマット
      display_format: "YYYY年MM月DD日",     // 表示フォーマット
    }
  } else {
    foramt = {
      month_format : "NNNNyoM月",           // 年月フォーマット
      edit_format : "NNyMMDD",              // 表示フォーマット
      display_format : "NNNNyoM月D日",      // 表示フォーマット
    }
  }
  document.querySelector("#sample4").datepicker(foramt);
}
